# Инициализация доски как массива 3x3
declare -A board
for i in {1..3}; do
    for j in {1..3}; do
        board[$i,$j]=" "
    done
done

# Функция для вывода доски
print_board() {
    echo -e "\n"
    for i in {1..3}; do
        echo " ${board[$i,1]} | ${board[$i,2]} | ${board[$i,3]} "
        if [ $i -lt 3 ]; then
            echo "-----------"
        fi
    done
    echo -e "\n"
}
